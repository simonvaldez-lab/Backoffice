    <script>
        function abrirModal(id) { document.getElementById(id).style.display = 'flex'; }
        function cerrarModal(id) { document.getElementById(id).style.display = 'none'; }
        
        function verificarOtros() {
            const select = document.getElementById('motivoRechazo');
            const campoOtros = document.getElementById('campoOtros');
            if(select.value === 'otros') { campoOtros.classList.remove('hidden'); } 
            else { campoOtros.classList.add('hidden'); }
        }

        // Abrir/Cerrar el panel de evidencias del costado derecho
        function alternarVisor() {
            const visor = document.getElementById('contenedorVisor');
            visor.style.display = (visor.style.display === 'flex') ? 'none' : 'flex';
        }

        // Cambiar entre la factura y la captura de pantalla del banco
        function cambiarPestana(origen) {
            const tabSol = document.getElementById('tabSol');
            const tabPrep = document.getElementById('tabPrep');
            const vistaSol = document.getElementById('vistaSolicitante');
            const vistaPrep = document.getElementById('vistaPreparador');

            if(origen === 'solicitante') {
                tabSol.classList.add('active');
                tabPrep.classList.remove('active');
                vistaSol.classList.remove('hidden');
                vistaPrep.classList.add('hidden');
            } else {
                tabPrep.classList.add('active');
                tabSol.classList.remove('active');
                vistaPrep.classList.remove('hidden');
                vistaSol.classList.add('hidden');
            }
        }

        // Lógica de simulación de traza final
        function ejecutarAccion(estado) {
            const ahora = new Date();
            const timestamp = ahora.toLocaleDateString() + ' - ' + ahora.toLocaleTimeString();
            
            const htmlTraza = `
                <strong>3. CIERRE DE FLUJO (${estado.toUpperCase()}):</strong><br>
                Procesado por: <span style="color:${estado === 'Aprobado' ? '#059669' : '#EF4444'}; font-weight:600;">Kate (Perfil Aprobador)</span><br>
                <span style="font-size:11px; color:#64748B;">Fecha y Hora: ${timestamp}</span>
            `;
            document.getElementById('trazaAprobador').innerHTML = htmlTraza;
            document.getElementById('trazaAprobador').style.background = estado === 'Aprobado' ? '#DCFCE7' : '#FEF2F2';

            alert(`Operación registrada como ${estado}.\nLa traza histórica guardó tu marca de tiempo (${timestamp}).\nNotificación enviada al preparador.`);
            
            cerrarModal('modalAprobar');
            cerrarModal('modalRechazo');
            abrirModal('modalDetalles');
        }

        function procesarExcel(evento, id) {
            const archivo = evento.target.files[0];
            const cajaResumen = document.getElementById(`resumen-excel-${id}`);
            const textoResumen = document.getElementById(`texto-resumen-${id}`);
            const inputRegistros = document.querySelector(`#solicitud-${id} .val-registros`);
            const inputValor = document.querySelector(`#solicitud-${id} .val-valor`);

            if (!archivo) {
                cajaResumen.style.display = 'none';
                return;
            }

            const lector = new FileReader();
            lector.onload = function(e) {
                const datos = new Uint8Array(e.target.result);
                const libro = XLSX.read(datos, { type: 'array' });
                const nombrePrimeraHoja = libro.SheetNames[0];
                const hoja = libro.Sheets[nombrePrimeraHoja];
                const filas = XLSX.utils.sheet_to_json(hoja);

                let cantidadFilas = filas.length;
                let valorTotal = 0;

                filas.forEach(fila => {
                    let valorFila = fila.Valor || fila.Monto || fila.valor || fila.monto || 0;
                    if (typeof valorFila === 'string') {
                        valorFila = parseFloat(valorFila.replace(/[^0-9.-]+/g, ""));
                    }
                    valorTotal += Number(valorFila) || 0;
                });

                cajaResumen.style.display = 'block';
                const formatoMoneda = new Intl.NumberFormat('es-CO', { style: 'currency', currency: 'COP' }).format(valorTotal);
                textoResumen.innerHTML = `Detectamos <b>${cantidadFilas} registros</b> por un total de <b>${formatoMoneda}</b>.`;

                if (inputRegistros) inputRegistros.value = cantidadFilas;
                if (inputValor) inputValor.value = valorTotal.toFixed(2);
            };
            lector.readAsArrayBuffer(archivo);
        }
